``` javascript
function Range(){}
Range.prototype = {
init:function(){
alert("XXX");
},
show:function(){
alert("YYY");
}
};

function RangeChildren(){}
RangeChildren.prototype = Range.prototype;
RangeChildren.prototype.add = function(){
alert("ADD");
};
```

调用：  
``` javascript
var rc = new RangeChildren();
rc.add();
rc.show();
```
